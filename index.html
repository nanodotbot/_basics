static methods

class Animal {
    constructor(picture, name) {
        this._picture = picture;
        this._name = name;
    }
    get picture() {
        return this._picture;
    }
    get name() {
        return this._name;
    }
}

class Cat extends Animal {
    constructor(picture, name, quietschelevel) {
        super(picture, name);
        this._quietschelevel = quietschelevel;
    }
    static meow() { // static method can only be called from class, not the instance
        return 'meow!';
    }
    get quietschelevel() {
        return this._quietschelevel;
    }
    incrementQuietschelevel() {
        this._quietschelevel++;
    }
}
const gato = new Cat('üêà', 'gato', 5);
